86    @Override
87    protected Shape parseShapeByType(WktShapeParser.State state, String shapeType) throws ParseException {
88      if (shapeType.equalsIgnoreCase("POLYGON")) {
89        return parsePolygonShape(state);
90      } else if (shapeType.equalsIgnoreCase("MULTIPOLYGON")) {
91        return parseMulitPolygonShape(state);
92      }
93      return super.parseShapeByType(state, shapeType);
94    }

96    /** Bypasses {@link JtsSpatialContext#makeLineString(java.util.List)} so that we can more
97     * efficiently get the LineString without creating a {@code List<Point>}.
98     */
99    @Override
100   protected Shape parseLineStringShape(WktShapeParser.State state) throws ParseException {
101     if (!ctx.useJtsLineString())
102       return super.parseLineStringShape(state);
103 
104     if (state.nextIfEmptyAndSkipZM())
105       return ctx.makeLineString(Collections.<Point>emptyList());
106 
107     GeometryFactory geometryFactory = ctx.getGeometryFactory();
108 
109     Coordinate[] coordinates = coordinateSequence(state);
110     return makeShapeFromGeometry(geometryFactory.createLineString(coordinates));
111   }
112 
113   /**
114    * Parses a POLYGON shape from the raw string. It might return a {@link com.spatial4j.core.shape.Rectangle}
115    * if the polygon is one.
116    * <pre>
117    *   coordinateSequenceList
118    * </pre>
119    */
120   protected Shape parsePolygonShape(WktShapeParser.State state) throws ParseException {
121     Geometry geometry;
122     if (state.nextIfEmptyAndSkipZM()) {
123       GeometryFactory geometryFactory = ctx.getGeometryFactory();
124       geometry = geometryFactory.createPolygon(geometryFactory.createLinearRing(
125           new Coordinate[]{}), null);
126     } else {
127       geometry = polygon(state);
128       if (geometry.isRectangle()) {
129         //TODO although, might want to never convert if there's a semantic difference (e.g. geodetically)
130         return makeRectFromPoly(geometry);
131       }
132     }
133     return makeShapeFromGeometry(geometry);
134   }
135 
136   protected Rectangle makeRectFromPoly(Geometry geometry) {
137     assert geometry.isRectangle();
138     Envelope env = geometry.getEnvelopeInternal();
139     boolean crossesDateline = false;
140     if (ctx.isGeo() && getDatelineRule() != DatelineRule.none) {
141       if (getDatelineRule() == DatelineRule.ccwRect) {
142         // If JTS says it is clockwise, then it's actually a dateline crossing rectangle.
143         crossesDateline = ! CGAlgorithms.isCCW(geometry.getCoordinates());
144       } else {
145         crossesDateline = env.getWidth() > 180;
146       }
147     }
148     if (crossesDateline)
149       return ctx.makeRectangle(env.getMaxX(), env.getMinX(), env.getMinY(), env.getMaxY());
150     else
151       return ctx.makeRectangle(env.getMinX(), env.getMaxX(), env.getMinY(), env.getMaxY());
152   }
153 
154   /**
155    * Reads a polygon, returning a JTS polygon.
156    */
157   protected Polygon polygon(WktShapeParser.State state) throws ParseException {
158     GeometryFactory geometryFactory = ctx.getGeometryFactory();
159 
160     List<Coordinate[]> coordinateSequenceList = coordinateSequenceList(state);
161 
162     LinearRing shell = geometryFactory.createLinearRing
163         (coordinateSequenceList.get(0));
164 
165     LinearRing[] holes = null;
166     if (coordinateSequenceList.size() > 1) {
167       holes = new LinearRing[coordinateSequenceList.size() - 1];
168       for (int i = 1; i < coordinateSequenceList.size(); i++) {
169         holes[i - 1] = geometryFactory.createLinearRing(coordinateSequenceList.get(i));
170       }
171     }
172     return geometryFactory.createPolygon(shell, holes);
173   }

175   /**
176    * Parses a MULTIPOLYGON shape from the raw string.
177    * <pre>
178    *   '(' polygon (',' polygon )* ')'
179    * </pre>
180    */
181   protected Shape parseMulitPolygonShape(WktShapeParser.State state) throws ParseException {
182     if (state.nextIfEmptyAndSkipZM())
183       return ctx.makeCollection(Collections.EMPTY_LIST);
184 
185     List<Shape> polygons = new ArrayList<Shape>();
186     state.nextExpect('(');
187     do {
188       polygons.add(parsePolygonShape(state));
189     } while (state.nextIf(','));
190     state.nextExpect(')');
191 
192     return ctx.makeCollection(polygons);
193   }

196   /**
197    * Reads a list of JTS Coordinate sequences from the current position.
198    * <pre>
199    *   '(' coordinateSequence (',' coordinateSequence )* ')'
200    * </pre>
201    */
202   protected List<Coordinate[]> coordinateSequenceList(WktShapeParser.State state) throws ParseException {
203     List<Coordinate[]> sequenceList = new ArrayList<Coordinate[]>();
204     state.nextExpect('(');
205     do {
206       sequenceList.add(coordinateSequence(state));
207     } while (state.nextIf(','));
208     state.nextExpect(')');
209     return sequenceList;
210   }
211 
212   /**
213    * Reads a JTS Coordinate sequence from the current position.
214    * <pre>
215    *   '(' coordinate (',' coordinate )* ')'
216    * </pre>
217    */
218   protected Coordinate[] coordinateSequence(WktShapeParser.State state) throws ParseException {
219     List<Coordinate> sequence = new ArrayList<Coordinate>();
220     state.nextExpect('(');
221     do {
222       sequence.add(coordinate(state));
223     } while (state.nextIf(','));
224     state.nextExpect(')');
225     return sequence.toArray(new Coordinate[sequence.size()]);
226   }

228   /**
229    * Reads a {@link com.vividsolutions.jts.geom.Coordinate} from the current position.
230    * It's akin to {@link #point(com.spatial4j.core.io.WktShapeParser.State)} but for
231    * a JTS Coordinate.  Only the first 2 numbers are parsed; any remaining are ignored.
232    */
233   protected Coordinate coordinate(WktShapeParser.State state) throws ParseException {
234     double x = ctx.normX(state.nextDouble());
235     ctx.verifyX(x);
236     double y = ctx.normY(state.nextDouble());
237     ctx.verifyY(y);
238     state.skipNextDoubles();
239     return new Coordinate(x, y);
240   }

242   @Override
243   protected double normDist(double v) {
244     return ctx.getGeometryFactory().getPrecisionModel().makePrecise(v);
245   }

247   /** Creates the JtsGeometry, potentially validating, repairing, and preparing. */
248   protected JtsGeometry makeShapeFromGeometry(Geometry geometry) {
249     final boolean dateline180Check = getDatelineRule() != DatelineRule.none;
250     JtsGeometry jtsGeom;
251     try {
252       jtsGeom = ctx.makeShape(geometry, dateline180Check, ctx.isAllowMultiOverlap());
253       if (isAutoValidate())
254         jtsGeom.validate();
255     } catch (RuntimeException e) {
256       //repair:
257       if (validationRule == ValidationRule.repairConvexHull) {
258         jtsGeom = ctx.makeShape(geometry.convexHull(), dateline180Check, ctx.isAllowMultiOverlap());
259       } else if (validationRule == ValidationRule.repairBuffer0) {
260         jtsGeom = ctx.makeShape(geometry.buffer(0), dateline180Check, ctx.isAllowMultiOverlap());
261       } else {
262         //TODO there are other smarter things we could do like repairing inner holes and subtracting
263         //  from outer repaired shell; but we needn't try too hard.
264         throw e;
265       }
266     }
267     if (isAutoIndex())
268       jtsGeom.index();
269     return jtsGeom;
270   }

272   /**
273    * Indicates the algorithm used to process JTS Polygons and JTS LineStrings for detecting dateline
274    * crossings. It only applies when geo=true.
275    */
276   public enum DatelineRule {
277     /** No polygon will cross the dateline. */
278     none,
279 
280     /** Adjacent points with an x (longitude) difference that spans more than half
281      * way around the globe will be interpreted as going the other (shorter) way, and thus cross the
282      * dateline.
283      */
284     width180,//TODO is there a better name that doesn't have '180' in it?
285 
286     /** For rectangular polygons, the point order is interpreted as being counter-clockwise (CCW).
287      * However, non-rectangular polygons or other shapes aren't processed this way; they use the
288      * {@link #width180} rule instead. The CCW rule is specified by OGC Simple Features
289      * Specification v. 1.2.0 section 6.1.11.1.
290      */
291     ccwRect
292   }
293 
294   /** Indicates how JTS geometries (notably polygons but applies to other geometries too) are
295    * validated (if at all) and repaired (if at all).
296    */
297   public enum ValidationRule {
298     /** Geometries will not be validated (because it's kinda expensive to calculate). You may or may
299      * not ultimately get an error at some point; results are undefined. However, note that
300      * coordinates will still be validated for falling within the world boundaries.
301      * @see com.vividsolutions.jts.geom.Geometry#isValid(). */
302     none,
303 
304     /** Geometries will be explicitly validated on creation, possibly resulting in an exception:
305      * {@link com.spatial4j.core.exception.InvalidShapeException}. */
306     error,
307 
308     /** Invalid Geometries are repaired by taking the convex hull. The result will very likely be a
309      * larger shape that matches false-positives, but no false-negatives.
310      * See {@link com.vividsolutions.jts.geom.Geometry#convexHull()}. */
311     repairConvexHull,
312 
313     /** Invalid polygons are repaired using the {@code buffer(0)} technique. From the <a
314      * href="http://tsusiatsoftware.net/jts/jts-faq/jts-faq.html">JTS FAQ</a>:
315      * <p>The buffer operation is fairly insensitive to topological invalidity, and the act of
316      * computing the buffer can often resolve minor issues such as self-intersecting rings. However,
317      * in some situations the computed result may not be what is desired (i.e. the buffer operation
318      * may be "confused" by certain topologies, and fail to produce a result which is close to the
319      * original. An example where this can happen is a "bow-tie: or "figure-8" polygon, with one
320      * very small lobe and one large one. Depending on the orientations of the lobes, the buffer(0)
321      * operation may keep the small lobe and discard the "valid" large lobe).
322      * </p> */
323     repairBuffer0
324   }
