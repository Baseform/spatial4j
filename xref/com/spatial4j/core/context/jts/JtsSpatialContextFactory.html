18  package com.spatial4j.core.context.jts;
19  
20  import com.spatial4j.core.context.SpatialContextFactory;
21  import com.spatial4j.core.io.jts.JtsBinaryCodec;
22  import com.spatial4j.core.io.jts.JtsWktShapeParser;
23  import com.vividsolutions.jts.geom.CoordinateSequenceFactory;
24  import com.vividsolutions.jts.geom.GeometryFactory;
25  import com.vividsolutions.jts.geom.PrecisionModel;
26  import com.vividsolutions.jts.geom.impl.CoordinateArraySequenceFactory;
27  
28  import java.util.Map;
29  
30  /**
31   * See {@link SpatialContextFactory#makeSpatialContext(java.util.Map, ClassLoader)}.
32   * <p/>
33   * The following keys are looked up in the args map, in addition to those in the
34   * superclass:
35   * <DL>
36   * <DT>datelineRule</DT>
37   * <DD>width180(default)|ccwRect|none
38   *  -- see {@link com.spatial4j.core.io.jts.JtsWktShapeParser.DatelineRule}</DD>
39   * <DT>validationRule</DT>
40   * <DD>error(default)|none|repairConvexHull|repairBuffer0
41   *  -- see {@link com.spatial4j.core.io.jts.JtsWktShapeParser.ValidationRule}</DD>
42   * <DT>autoIndex</DT>
43   * <DD>true|false(default) -- see {@link JtsWktShapeParser#isAutoIndex()}</DD>
44   * <DT>allowMultiOverlap</DT>
45   * <DD>true|false(default) -- see {@link JtsSpatialContext#isAllowMultiOverlap()}</DD>
46   * <DT>precisionModel</DT>
47   * <DD>floating(default) | floating_single | fixed
48   *  -- see {@link com.vividsolutions.jts.geom.PrecisionModel}.
49   * If {@code fixed} then you must also provide {@code precisionScale}
50   *  -- see {@link com.vividsolutions.jts.geom.PrecisionModel#getScale()}</DD>
51   * </DL>
52   */
53  public class JtsSpatialContextFactory extends SpatialContextFactory {
54  
55    protected static final PrecisionModel defaultPrecisionModel = new PrecisionModel();//floating
56  
57    //These 3 are JTS defaults for new GeometryFactory()
58    public PrecisionModel precisionModel = defaultPrecisionModel;
59    public int srid = 0;
60    public CoordinateSequenceFactory coordinateSequenceFactory = CoordinateArraySequenceFactory.instance();
61  
62    //ignored if geo=false
63    public JtsWktShapeParser.DatelineRule datelineRule = JtsWktShapeParser.DatelineRule.width180;
64  
65    public JtsWktShapeParser.ValidationRule validationRule = JtsWktShapeParser.ValidationRule.error;
66    public boolean autoIndex = false;
67    public boolean allowMultiOverlap = false;//ignored if geo=false
68  
69    //kinda advanced options:
70    public boolean useJtsPoint = true;
71    public boolean useJtsLineString = true;
72  
73    public JtsSpatialContextFactory() {
74      super.wktShapeParserClass = JtsWktShapeParser.class;
75      super.binaryCodecClass = JtsBinaryCodec.class;
76    }
77    
78    @Override
79    protected void init(Map<String, String> args, ClassLoader classLoader) {
80      super.init(args, classLoader);
81  
82      initField("datelineRule");
83      initField("validationRule");
84      initField("autoIndex");
85      initField("allowMultiOverlap");
86      initField("useJtsPoint");
87      initField("useJtsLineString");
88  
89      String scaleStr = args.get("precisionScale");
90      String modelStr = args.get("precisionModel");
91  
92      if (scaleStr != null) {
93        if (modelStr != null && !modelStr.equals("fixed"))
94          throw new RuntimeException("Since precisionScale was specified; precisionModel must be 'fixed' but got: "+modelStr);
95        precisionModel = new PrecisionModel(Double.parseDouble(scaleStr));
96      } else if (modelStr != null) {
97        if (modelStr.equals("floating")) {
98          precisionModel = new PrecisionModel(PrecisionModel.FLOATING);
99        } else if (modelStr.equals("floating_single")) {
100         precisionModel = new PrecisionModel(PrecisionModel.FLOATING_SINGLE);
101       } else if (modelStr.equals("fixed")) {
102         throw new RuntimeException("For fixed model, must specifiy 'precisionScale'");
103       } else {
104         throw new RuntimeException("Unknown precisionModel: "+modelStr);
105       }
106     }
107   }
108 
109   public GeometryFactory getGeometryFactory() {
110     if (precisionModel == null || coordinateSequenceFactory == null)
111       throw new IllegalStateException("precision model or coord seq factory can't be null");
112     return new GeometryFactory(precisionModel, srid, coordinateSequenceFactory);
113   }
114 
115   @Override
116   public JtsSpatialContext newSpatialContext() {
117     return new JtsSpatialContext(this);
118   }
119 }
