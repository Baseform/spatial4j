35  /**
36   * Enhances the default {@link SpatialContext} with support for Polygons (and
37   * other geometries) using <a href="https://sourceforge.net/projects/jts-topo-suite/">JTS</a>.
38   * To the extent possible, our {@link JtsGeometry} adds some amount of geodetic support over
39   * vanilla JTS which only has a Euclidean (flat plane) model.
40   */
41  public class JtsSpatialContext extends SpatialContext {
42  
43    public static final JtsSpatialContext GEO;
44    static {
45      JtsSpatialContextFactory factory = new JtsSpatialContextFactory();
46      factory.geo = true;
47      GEO = new JtsSpatialContext(factory);
48    }
49  
50    protected final GeometryFactory geometryFactory;
51  
52    protected final boolean allowMultiOverlap;
53    protected final boolean useJtsPoint;
54    protected final boolean useJtsLineString;

56    /**
57     * Called by {@link com.spatial4j.core.context.jts.JtsSpatialContextFactory#newSpatialContext()}.
58     */
59    public JtsSpatialContext(JtsSpatialContextFactory factory) {
60      super(factory);
61      this.geometryFactory = factory.getGeometryFactory();
62  
63      this.allowMultiOverlap = factory.allowMultiOverlap;
64      this.useJtsPoint = factory.useJtsPoint;
65      this.useJtsLineString = factory.useJtsLineString;
66    }

68    /**
69     * If geom might be a multi geometry of some kind, then might multiple
70     * component geometries overlap? Strict OGC says this is invalid but we
71     * can accept it by computing the union. Note: Our ShapeCollection mostly
72     * doesn't care but it has a method related to this
73     * {@link com.spatial4j.core.shape.ShapeCollection#relateContainsShortCircuits()}.
74     */
75    public boolean isAllowMultiOverlap() {
76      return allowMultiOverlap;
77    }

91    @Override
92    public String toString(Shape shape) {
93      //Note: this logic is from the defunct JtsShapeReadWriter
94      if (shape instanceof JtsGeometry) {
95        JtsGeometry jtsGeom = (JtsGeometry) shape;
96        return jtsGeom.getGeom().toText();
97      }
98      //Note: doesn't handle ShapeCollection or BufferedLineString
99      return super.toString(shape);
100   }

102   /**
103    * Gets a JTS {@link Geometry} for the given {@link Shape}. Some shapes hold a
104    * JTS geometry whereas new ones must be created for the rest.
105    * @param shape Not null
106    * @return Not null
107    */
108   public Geometry getGeometryFrom(Shape shape) {
109     if (shape instanceof JtsGeometry) {
110       return ((JtsGeometry)shape).getGeom();
111     }
112     if (shape instanceof JtsPoint) {
113       return ((JtsPoint) shape).getGeom();
114     }
115     if (shape instanceof Point) {
116       Point point = (Point) shape;
117       return geometryFactory.createPoint(new Coordinate(point.getX(),point.getY()));
118     }
119     if (shape instanceof Rectangle) {
120       Rectangle r = (Rectangle)shape;
121       if (r.getCrossesDateLine()) {
122         Collection<Geometry> pair = new ArrayList<Geometry>(2);
123         pair.add(geometryFactory.toGeometry(new Envelope(
124                 r.getMinX(), getWorldBounds().getMaxX(), r.getMinY(), r.getMaxY())));
125         pair.add(geometryFactory.toGeometry(new Envelope(
126                 getWorldBounds().getMinX(), r.getMaxX(), r.getMinY(), r.getMaxY())));
127         return geometryFactory.buildGeometry(pair);//a MultiPolygon or MultiLineString
128       } else {
129         return geometryFactory.toGeometry(new Envelope(r.getMinX(), r.getMaxX(), r.getMinY(), r.getMaxY()));
130       }
131     }
132     if (shape instanceof Circle) {
133       // FYI Some interesting code for this is here:
134       //  http://docs.codehaus.org/display/GEOTDOC/01+How+to+Create+a+Geometry#01HowtoCreateaGeometry-CreatingaCircle
135       //TODO This should ideally have a geodetic version
136       Circle circle = (Circle)shape;
137       if (circle.getBoundingBox().getCrossesDateLine())
138         throw new IllegalArgumentException("Doesn't support dateline cross yet: "+circle);//TODO
139       GeometricShapeFactory gsf = new GeometricShapeFactory(geometryFactory);
140       gsf.setSize(circle.getBoundingBox().getWidth());
141       gsf.setNumPoints(4*25);//multiple of 4 is best
142       gsf.setCentre(new Coordinate(circle.getCenter().getX(), circle.getCenter().getY()));
143       return gsf.createCircle();
144     }
145     //TODO add BufferedLineString
146     throw new InvalidShapeException("can't make Geometry from: " + shape);
147   }

149   /** Should {@link #makePoint(double, double)} return {@link JtsPoint}? */
150   public boolean useJtsPoint() {
151     return useJtsPoint;
152   }
153 
154   @Override
155   public Point makePoint(double x, double y) {
156     if (!useJtsPoint())
157       return super.makePoint(x, y);
158     //A Jts Point is fairly heavyweight!  TODO could/should we optimize this? SingleCoordinateSequence
159     verifyX(x);
160     verifyY(y);
161     Coordinate coord = Double.isNaN(x) ? null : new Coordinate(x, y);
162     return new JtsPoint(geometryFactory.createPoint(coord), this);
163   }
164 
165   /** Should {@link #makeLineString(java.util.List)} return {@link JtsGeometry}? */
166   public boolean useJtsLineString() {
167     //BufferedLineString doesn't yet do dateline cross, and can't yet be relate()'ed with a
168     // JTS geometry
169     return useJtsLineString;
170   }
171 
172   @Override
173   public Shape makeLineString(List<Point> points) {
174     if (!useJtsLineString())
175       return super.makeLineString(points);
176     //convert List<Point> to Coordinate[]
177     Coordinate[] coords = new Coordinate[points.size()];
178     for (int i = 0; i < coords.length; i++) {
179       Point p = points.get(i);
180       if (p instanceof JtsPoint) {
181         JtsPoint jtsPoint = (JtsPoint) p;
182         coords[i] = jtsPoint.getGeom().getCoordinate();
183       } else {
184         coords[i] = new Coordinate(p.getX(), p.getY());
185       }
186     }
187     LineString lineString = geometryFactory.createLineString(coords);
188     return makeShape(lineString);
189   }

191   /**
192    * INTERNAL
193    * @see #makeShape(com.vividsolutions.jts.geom.Geometry)
194    *
195    * @param geom Non-null
196    * @param dateline180Check if both this is true and {@link #isGeo()}, then JtsGeometry will check
197    *                         for adjacent coordinates greater than 180 degrees longitude apart, and
198    *                         it will do tricks to make that line segment (and the shape as a whole)
199    *                         cross the dateline even though JTS doesn't have geodetic support.
200    * @param allowMultiOverlap See {@link #isAllowMultiOverlap()}.
201    */
202   public JtsGeometry makeShape(Geometry geom, boolean dateline180Check, boolean allowMultiOverlap) {
203     return new JtsGeometry(geom, this, dateline180Check, allowMultiOverlap);
204   }
205 
206   /**
207    * INTERNAL: Creates a {@link Shape} from a JTS {@link Geometry}. Generally, this shouldn't be
208    * called when one of the other factory methods are available, such as for points. The caller
209    * needs to have done some verification/normalization of the coordinates by now, if any.
210    */
211   public JtsGeometry makeShape(Geometry geom) {
212     return makeShape(geom, true/*dateline180Check*/, allowMultiOverlap);
213   }

215   public GeometryFactory getGeometryFactory() {
216     return geometryFactory;
217   }

219   @Override
220   public String toString() {
221     if (this.equals(GEO)) {
222       return GEO.getClass().getSimpleName()+".GEO";
223     } else {
224       return super.toString();
225     }
226   }
