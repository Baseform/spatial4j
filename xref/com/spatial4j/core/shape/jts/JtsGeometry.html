54   * Wraps a JTS {@link Geometry} (i.e. may be a polygon or basically anything).
55   * JTS does a great deal of the hard work, but there is work here in handling
56   * dateline wrap.

69    public JtsGeometry(Geometry geom, JtsSpatialContext ctx, boolean dateline180Check, boolean allowMultiOverlap) {
70      this.ctx = ctx;
71      //GeometryCollection isn't supported in relate()
72      if (geom.getClass().equals(GeometryCollection.class))
73        throw new IllegalArgumentException("JtsGeometry does not support GeometryCollection but does support its subclasses.");
74  
75      //NOTE: All this logic is fairly expensive. There are some short-circuit checks though.
76      if (ctx.isGeo()) {
77        //Unwraps the geometry across the dateline so it exceeds the standard geo bounds (-180 to +180).
78        if (dateline180Check)
79          unwrapDateline(geom);//potentially modifies geom
80        //If given multiple overlapping polygons, fix it by union
81        if (allowMultiOverlap)
82          geom = unionGeometryCollection(geom);//returns same or new geom
83  
84        //Cuts an unwrapped geometry back into overlaid pages in the standard geo bounds.
85        geom = cutUnwrappedGeomInto360(geom);//returns same or new geom
86        assert geom.getEnvelopeInternal().getWidth() <= 360;
87        assert ! geom.getClass().equals(GeometryCollection.class) : "GeometryCollection unsupported";//double check
88  
89        //Compute bbox
90        bbox = computeGeoBBox(geom);
91      } else {//not geo
92        //If given multiple overlapping polygons, fix it by union
93        if (allowMultiOverlap)
94          geom = unionGeometryCollection(geom);//returns same or new geom
95  
96        Envelope env = geom.getEnvelopeInternal();
97        bbox = new RectangleImpl(env.getMinX(), env.getMaxX(), env.getMinY(), env.getMaxY(), ctx);
98      }
99      geom.getEnvelopeInternal();//ensure envelope is cached internally, which is lazy evaluated. Keeps this thread-safe.
100 
101     this.geom = geom;
102     assert assertValidate();//kinda expensive but caches valid state
103 
104     this.hasArea = !((geom instanceof Lineal) || (geom instanceof Puntal));
105   }

107   /** called via assertion */
108   private boolean assertValidate() {
109     String assertValidate = System.getProperty(SYSPROP_ASSERT_VALIDATE);
110     if (assertValidate == null || Boolean.parseBoolean(assertValidate))
111       validate();
112     return true;
113   }

115   /**
116    * Validates the shape, throwing a descriptive error if it isn't valid. Note that this
117    * is usually called automatically by default, but that can be disabled.
118    *
119    * @throws InvalidShapeException with descriptive error if the shape isn't valid
120    */
121   public void validate() throws InvalidShapeException {
122     if (!validated) {
123       IsValidOp isValidOp = new IsValidOp(geom);
124       if (!isValidOp.isValid())
125         throw new InvalidShapeException(isValidOp.getValidationError().toString());
126       validated = true;
127     }
128   }

135    * already indexed then nothing happens.

138     if (preparedGeometry == null)
139       preparedGeometry = PreparedGeometryFactory.prepare(geom);

147   /** Given {@code geoms} which has already been checked for being in world
148    * bounds, return the minimal longitude range of the bounding box.
149    */
150   protected Rectangle computeGeoBBox(Geometry geoms) {
151     if (geoms.isEmpty())
152       return new RectangleImpl(Double.NaN, Double.NaN, Double.NaN, Double.NaN, ctx);
153     final Envelope env = geoms.getEnvelopeInternal();//for minY & maxY (simple)
154     if (env.getWidth() > 180 && geoms.getNumGeometries() > 1)  {
155       // This is ShapeCollection's bbox algorithm
156       Range xRange = null;
157       for (int i = 0; i < geoms.getNumGeometries(); i++ ) {
158         Envelope envI = geoms.getGeometryN(i).getEnvelopeInternal();
159         Range xRange2 = new Range.LongitudeRange(envI.getMinX(), envI.getMaxX());
160         if (xRange == null) {
161           xRange = xRange2;
162         } else {
163           xRange = xRange.expandTo(xRange2);
164         }
165         if (xRange == Range.LongitudeRange.WORLD_180E180W)
166           break; // can't grow any bigger
167       }
168       return new RectangleImpl(xRange.getMin(), xRange.getMax(), env.getMinY(), env.getMaxY(), ctx);
169     } else {
170       return new RectangleImpl(env.getMinX(), env.getMaxX(), env.getMinY(), env.getMaxY(), ctx);
171     }
172   }

186   @Override
187   public double getArea(SpatialContext ctx) {
188     double geomArea = geom.getArea();
189     if (ctx == null || geomArea == 0)
190       return geomArea;
191     //Use the area proportional to how filled the bbox is.
192     double bboxArea = getBoundingBox().getArea(null);//plain 2d area
193     assert bboxArea >= geomArea;
194     double filledRatio = geomArea / bboxArea;
195     return getBoundingBox().getArea(ctx) * filledRatio;
196     // (Future: if we know we use an equal-area projection then we don't need to
197     //  estimate)
198   }

212   @Override
213   public SpatialRelation relate(Shape other) {
214     if (other instanceof Point)
215       return relate((Point)other);
216     else if (other instanceof Rectangle)
217       return relate((Rectangle) other);
218     else if (other instanceof Circle)
219       return relate((Circle) other);
220     else if (other instanceof JtsGeometry)
221       return relate((JtsGeometry) other);
222     else if (other instanceof BufferedLineString)
223       throw new UnsupportedOperationException("Can't use BufferedLineString with JtsGeometry");
224     return other.relate(this).transpose();
225   }

227   public SpatialRelation relate(Point pt) {
228     if (!getBoundingBox().relate(pt).intersects())
229       return SpatialRelation.DISJOINT;
230     Geometry ptGeom;
231     if (pt instanceof JtsPoint)
232       ptGeom = ((JtsPoint)pt).getGeom();
233     else
234       ptGeom = ctx.getGeometryFactory().createPoint(new Coordinate(pt.getX(), pt.getY()));
235     return relate(ptGeom);//is point-optimized
236   }

238   public SpatialRelation relate(Rectangle rectangle) {
239     SpatialRelation bboxR = bbox.relate(rectangle);
240     if (bboxR == SpatialRelation.WITHIN || bboxR == SpatialRelation.DISJOINT)
241       return bboxR;
242     // FYI, the right answer could still be DISJOINT or WITHIN, but we don't know yet.
243     return relate(ctx.getGeometryFrom(rectangle));
244   }

246   public SpatialRelation relate(Circle circle) {
247     SpatialRelation bboxR = bbox.relate(circle);
248     if (bboxR == SpatialRelation.WITHIN || bboxR == SpatialRelation.DISJOINT)
249       return bboxR;
250 
251     //Test each point to see how many of them are outside of the circle.
252     //TODO consider instead using geom.apply(CoordinateSequenceFilter) -- maybe faster since avoids Coordinate[] allocation
253     Coordinate[] coords = geom.getCoordinates();
254     int outside = 0;
255     int i = 0;
256     for (Coordinate coord : coords) {
257       i++;
258       SpatialRelation sect = circle.relate(new PointImpl(coord.x, coord.y, ctx));
259       if (sect == SpatialRelation.DISJOINT)
260         outside++;
261       if (i != outside && outside != 0)//short circuit: partially outside, partially inside
262         return SpatialRelation.INTERSECTS;
263     }
264     if (i == outside) {
265       return (relate(circle.getCenter()) == SpatialRelation.DISJOINT)
266           ? SpatialRelation.DISJOINT : SpatialRelation.CONTAINS;
267     }
268     assert outside == 0;
269     return SpatialRelation.WITHIN;
270   }

277   protected SpatialRelation relate(Geometry oGeom) {
278     //see http://docs.geotools.org/latest/userguide/library/jts/dim9.html#preparedgeometry
279     if (oGeom instanceof com.vividsolutions.jts.geom.Point) {
280       if (preparedGeometry != null)
281         return preparedGeometry.disjoint(oGeom) ? SpatialRelation.DISJOINT : SpatialRelation.CONTAINS;
282       return geom.disjoint(oGeom) ? SpatialRelation.DISJOINT : SpatialRelation.CONTAINS;
283     }
284     if (preparedGeometry == null)
285       return intersectionMatrixToSpatialRelation(geom.relate(oGeom));
286     else if (preparedGeometry.covers(oGeom))
287       return SpatialRelation.CONTAINS;
288     else if (preparedGeometry.coveredBy(oGeom))
289       return SpatialRelation.WITHIN;
290     else if (preparedGeometry.intersects(oGeom))
291       return SpatialRelation.INTERSECTS;
292     return SpatialRelation.DISJOINT;
293   }

295   public static SpatialRelation intersectionMatrixToSpatialRelation(IntersectionMatrix matrix) {
296     //As indicated in SpatialRelation javadocs, Spatial4j CONTAINS & WITHIN are
297     // OGC's COVERS & COVEREDBY
298     if (matrix.isCovers())
299       return SpatialRelation.CONTAINS;
300     else if (matrix.isCoveredBy())
301       return SpatialRelation.WITHIN;
302     else if (matrix.isDisjoint())
303       return SpatialRelation.DISJOINT;
304     return SpatialRelation.INTERSECTS;
305   }

330   /**
331    * If <code>geom</code> spans the dateline, then this modifies it to be a
332    * valid JTS geometry that extends to the right of the standard -180 to +180
333    * width such that some points are greater than +180 but some remain less.
334    * Takes care to invoke {@link com.vividsolutions.jts.geom.Geometry#geometryChanged()}
335    * if needed.
336    *
337    * @return The number of times the geometry spans the dateline.  >= 0
338    */
339   private static int unwrapDateline(Geometry geom) {
340     if (geom.getEnvelopeInternal().getWidth() < 180)
341       return 0;//can't possibly cross the dateline
342     final int[] crossings = {0};//an array so that an inner class can modify it.
343     geom.apply(new GeometryFilter() {
344       @Override
345       public void filter(Geometry geom) {
346         int cross = 0;
347         if (geom instanceof LineString) {//note: LinearRing extends LineString
348           if (geom.getEnvelopeInternal().getWidth() < 180)
349             return;//can't possibly cross the dateline
350           cross = unwrapDateline((LineString) geom);
351         } else if (geom instanceof Polygon) {
352           if (geom.getEnvelopeInternal().getWidth() < 180)
353             return;//can't possibly cross the dateline
354           cross = unwrapDateline((Polygon) geom);
355         } else
356           return;
357         crossings[0] = Math.max(crossings[0], cross);
358       }
359     });//geom.apply()
360 
361     return crossings[0];
362   }

364   /** See {@link #unwrapDateline(Geometry)}. */
365   private static int unwrapDateline(Polygon poly) {
366     LineString exteriorRing = poly.getExteriorRing();
367     int cross = unwrapDateline(exteriorRing);
368     if (cross > 0) {
369       //TODO TEST THIS! Maybe bug if doesn't cross but is in another page?
370       for(int i = 0; i < poly.getNumInteriorRing(); i++) {
371         LineString innerLineString = poly.getInteriorRingN(i);
372         unwrapDateline(innerLineString);
373         for(int shiftCount = 0; ! exteriorRing.contains(innerLineString); shiftCount++) {
374           if (shiftCount > cross)
375             throw new IllegalArgumentException("The inner ring doesn't appear to be within the exterior: "
376                 +exteriorRing+" inner: "+innerLineString);
377           shiftGeomByX(innerLineString, 360);
378         }
379       }
380       poly.geometryChanged();
381     }
382     return cross;
383   }

385   /** See {@link #unwrapDateline(Geometry)}. */
386   private static int unwrapDateline(LineString lineString) {
387     CoordinateSequence cseq = lineString.getCoordinateSequence();
388     int size = cseq.size();
389     if (size <= 1)
390       return 0;
391 
392     int shiftX = 0;//invariant: == shiftXPage*360
393     int shiftXPage = 0;
394     int shiftXPageMin = 0/* <= 0 */, shiftXPageMax = 0; /* >= 0 */
395     double prevX = cseq.getX(0);
396     for(int i = 1; i < size; i++) {
397       double thisX_orig = cseq.getX(i);
398       assert thisX_orig >= -180 && thisX_orig <= 180 : "X not in geo bounds";
399       double thisX = thisX_orig + shiftX;
400       if (prevX - thisX > 180) {//cross dateline from left to right
401         thisX += 360;
402         shiftX += 360;
403         shiftXPage += 1;
404         shiftXPageMax = Math.max(shiftXPageMax,shiftXPage);
405       } else if (thisX - prevX > 180) {//cross dateline from right to left
406         thisX -= 360;
407         shiftX -= 360;
408         shiftXPage -= 1;
409         shiftXPageMin = Math.min(shiftXPageMin,shiftXPage);
410       }
411       if (shiftXPage != 0)
412         cseq.setOrdinate(i, CoordinateSequence.X, thisX);
413       prevX = thisX;
414     }
415     if (lineString instanceof LinearRing) {
416       assert cseq.getCoordinate(0).equals(cseq.getCoordinate(size-1));
417       assert shiftXPage == 0;//starts and ends at 0
418     }
419     assert shiftXPageMax >= 0 && shiftXPageMin <= 0;
420     //Unfortunately we are shifting again; it'd be nice to be smarter and shift once
421     shiftGeomByX(lineString, shiftXPageMin * -360);
422     int crossings = shiftXPageMax - shiftXPageMin;
423     if (crossings > 0)
424       lineString.geometryChanged();
425     return crossings;
426   }

428   private static void shiftGeomByX(Geometry geom, final int xShift) {
429     if (xShift == 0)
430       return;
431     geom.apply(new CoordinateSequenceFilter() {
432       @Override
433       public void filter(CoordinateSequence seq, int i) {
434         seq.setOrdinate(i, CoordinateSequence.X, seq.getX(i) + xShift );
435       }
436 
437       @Override public boolean isDone() { return false; }
438 
439       @Override public boolean isGeometryChanged() { return true; }
440     });
441   }

450   /**
451    * This "pages" through standard geo boundaries offset by multiples of 360
452    * longitudinally that intersect geom, and the intersecting results of a page
453    * and the geom are shifted into the standard -180 to +180 and added to a new
454    * geometry that is returned.
455    */
456   private static Geometry cutUnwrappedGeomInto360(Geometry geom) {
457     Envelope geomEnv = geom.getEnvelopeInternal();
458     if (geomEnv.getMinX() >= -180 && geomEnv.getMaxX() <= 180)
459       return geom;
460     assert geom.isValid() : "geom";
461 
462     //TODO opt: support geom's that start at negative pages --
463     // ... will avoid need to previously shift in unwrapDateline(geom).
464     List<Geometry> geomList = new ArrayList<Geometry>();
465     //page 0 is the standard -180 to 180 range
466     for (int page = 0; true; page++) {
467       double minX = -180 + page * 360;
468       if (geomEnv.getMaxX() <= minX)
469         break;
470       Geometry rect = geom.getFactory().toGeometry(new Envelope(minX, minX + 360, -90, 90));
471       assert rect.isValid() : "rect";
472       Geometry pageGeom = rect.intersection(geom);//JTS is doing some hard work
473       assert pageGeom.isValid() : "pageGeom";
474 
475       shiftGeomByX(pageGeom, page * -360);
476       geomList.add(pageGeom);
477     }
478     return UnaryUnionOp.union(geomList);
479   }
